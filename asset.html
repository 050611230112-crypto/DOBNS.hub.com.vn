// AUTO LOAD YEAR vào dropdown window.onload = function () { const table = document.getElementById("financeTable"); const rows = table.getElementsByTagName("tr"); const yearSet = new Set(); for (let i = 1; i < rows.length; i++) { const dateText = rows[i].cells[0].innerText; const year = "20" + dateText.split("/")[2]; yearSet.add(year); } const select = document.getElementById("yearFilter"); yearSet.forEach(year => { const option = document.createElement("option"); option.value = year; option.text = year; select.appendChild(option); }); }; // FILTER function filterYear() { const selectedYear = document.getElementById("yearFilter").value; const rows = document.querySelectorAll("#financeTable tbody tr"); rows.forEach(row => { const dateText = row.cells[0].innerText; const year = "20" + dateText.split("/")[2]; if (selectedYear === "all" || year === selectedYear) { row.style.display = ""; } else { row.style.display = "none"; } }); }
